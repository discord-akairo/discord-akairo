util/Category.js

const { Collection } = require('discord.js');

/** @extends Collection */
class Category extends Collection {
    /**
     * A group of modules.
     * @param {string} id - ID of the category.
     */
    constructor(id){
        super();

        /**
         * ID of the category.
         * @type {string}
         */
        this.id = id;
    }

    /**
     * Calls reload() on all items in this category.
     * @returns {Category}
     */
    reloadAll(){
        for (const m of Array.from(this.values())) m.reload();
        return this;
    }

    /**
     * Calls remove() on all items in this category.
     * @returns {Category}
     */
    removeAll(){
        for (const m of Array.from(this.values())) m.remove();
        return this;
    }

    /**
     * Calls enable() on all items in this category.
     * @returns {Category}
     */
    enableAll(){
        for (const m of this.values()) m.enable();
        return this;
    }

    /**
     * Calls disable() on all items in this category.
     * @returns {Category}
     */
    disableAll(){
        for (const m of this.values()) m.disable();
        return this;
    }

    /**
     * Returns the ID.
     * @returns {string}
     */
    toString(){
        return this.id;
    }
}

module.exports = Category;
